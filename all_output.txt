===== ./README.md =====
## Export NetBox ra Excel Master

Script `scripts/export_netbox_to_excel.py` keo du lieu tu NetBox API va xuat ra file Excel nhieu sheet:

- `Inventory`
- `IP_Seg`
- `Prefixes`
- `Maport`
- `IP_Addresses`
- `Cables`

Cai thu vien:

```bash
python3 -m pip install -r requirements.txt
```

Cai Ansible collection cho Aruba AOS-CX:

```bash
ansible-galaxy collection install -r collections/requirements.yml
```

Chay export voi cau hinh NetBox dang co trong `inventories/lab/netbox_inventory.yml`:

```bash
python3 scripts/export_netbox_to_excel.py
```

Hoac truyen URL/token rieng:

```bash
NETBOX_URL=http://localhost:8000 NETBOX_TOKEN=your_token python3 scripts/export_netbox_to_excel.py
```

File mac dinh se duoc tao tai:

```text
generated_reports/NetBox_Master_Export.xlsx
```

## Import Excel Master vao NetBox

Script `scripts/import_to_netbox.py` doc file Excel master va day du lieu len NetBox theo dung thu tu dependency:

1. Regions, Sites, Locations, Racks
2. VLANs, Prefixes
3. Manufacturers, Device Roles, Device Types
4. Devices, Interfaces
5. IP Addresses, Primary IP, Cables

File mac dinh:

```text
data/Master_Infra_Config.xlsx
```

Neu file tren chua ton tai, script se fallback sang workbook hien co:

```text
data/Infra_config.xlsx
```

Script cung fallback sang cac CSV cu trong `data/` neu workbook bi thieu sheet.

Cau hinh token an toan:

```bash
cp scripts/.env.example scripts/.env
```

Sau do sua `scripts/.env`:

```text
NETBOX_URL=http://localhost:8000
NETBOX_TOKEN=your_token
```

Kiem tra workbook/CSV truoc khi import:

```bash
python3 scripts/import_to_netbox.py --validate-only
```

Chay thu voi NetBox nhung chi in hanh dong:

```bash
python3 scripts/import_to_netbox.py --dry-run
```

Import that:

```bash
python3 scripts/import_to_netbox.py
```

    target_site: "Lab Hanoi"
    target_role: "access_switches"

  tasks:

    # PHẦN 1: BỎ QUA FACTS, CHỈ CÀO FILE CONFIG
    # ==========================================
    - name: 1. Đọc File Running-Config trực tiếp qua CLI
      cisco.ios.ios_command:
        commands: "show running-config"
      register: switch_config

    - name: 2. Lưu file cấu hình gốc ra thư mục backups
      copy:
        # Lấy kết quả đầu ra của lệnh đầu tiên (show run)
        content: "{{ switch_config.stdout[0] }}"
        dest: "{{ playbook_dir }}/../backups/{{ inventory_hostname }}_original_backup.cfg"
      delegate_to: localhost

    - name: 3. Hiển thị thông báo
      debug:
        msg: "Đã cào file cấu hình thành công! Bắt đầu tạo NetBox..."

    # ==========================================
    # PHẦN 2: ĐẨY DỮ LIỆU LÊN NETBOX
    # ==========================================
    - name: 4. Đảm bảo Loại thiết bị (Device Type) có tồn tại trên NetBox
      netbox.netbox.netbox_device_type:
        netbox_url: "{{ netbox_url }}"
        netbox_token: "{{ netbox_token }}"
        data:
          manufacturer: "Cisco"
          model: "C1200-24T-4X"
        state: present
      delegate_to: localhost
    # ---> THÊM ĐOẠN NÀY VÀO ĐÂY <---
    - name: 4b. Đảm bảo Vai trò (Device Role) có tồn tại trên NetBox
      netbox.netbox.netbox_device_role:
        netbox_url: "{{ netbox_url }}"
        netbox_token: "{{ netbox_token }}"
        data:
          name: "{{ target_role }}"
          color: "00bcd4"  # Màu xanh nước biển cho Access Switch
        state: present
      delegate_to: localhost
    # ---------------------------------
    - name: 5. Khởi tạo con Switch SW_BACKMA trên NetBox
      netbox.netbox.netbox_device:
        netbox_url: "{{ netbox_url }}"
        netbox_token: "{{ netbox_token }}"
        data:
          name: "{{ inventory_hostname }}"
          device_type: "C1200-24T-4X"
          device_role: "{{ target_role }}"
          site: "{{ target_site }}"
          status: "active"
        state: present
      delegate_to: localhost

    - name: 6a. Tự động tạo 24 Port GigabitEthernet (Đồng) bằng vòng lặp
      netbox.netbox.netbox_device_interface:
        netbox_url: "{{ netbox_url }}"
        netbox_token: "{{ netbox_token }}"
        data:
          device: "{{ inventory_hostname }}"
          name: "gi{{ item }}"  # Đặt tên theo chuẩn C1200 (gi1 -> gi24)
          type: "1000base-t"
        state: present
      loop: "{{ range(1, 25) | list }}"
      delegate_to: localhost

    - name: 6b. Tự động tạo 4 Port 10G SFP+ (Quang) bằng vòng lặp
      netbox.netbox.netbox_device_interface:
        netbox_url: "{{ netbox_url }}"
        netbox_token: "{{ netbox_token }}"
        data:
          device: "{{ inventory_hostname }}"
          name: "te{{ item }}"  # Đặt tên cổng quang (te1 -> te4)
          type: "10gbase-x-sfpp"
        state: present
      loop: "{{ range(1, 5) | list }}"
      delegate_to: localhost

    - name: 7. Gán địa chỉ IP Quản trị cho Switch trên NetBox
      netbox.netbox.netbox_ip_address:
        netbox_url: "{{ netbox_url }}"
        netbox_token: "{{ netbox_token }}"
        data:
          address: "{{ ansible_host }}/24" 
          status: "active"

            device: "{{ inventory_hostname }}"
            name: "gi1" # Tạm gán vào cổng 1, có thể lên Web đổi sang Vlan1 sau

      delegate_to: localhost
      ignore_errors: yes

===== ./playbooks/sync_netbox_to_switches.yml =====
---
- name: ĐỒNG BỘ CẤU HÌNH TỪ NETBOX XUỐNG THIẾT BỊ MẠNG
  hosts: all  # Đảm bảo bạn đã map group này trong netbox_inventory.yml
  gather_facts: no
  vars:
    # Cấu hình API kết nối đến NetBox của bạn
    netbox_url: "http://localhost:8000"
    netbox_token: "nbt_U50yW0FdkGaq.Jw5AEEphCT94JwvvHfdOIdGOGYcn8dWanuNnQWNw"
    cisco_network_os_values:
      - ios
      - cisco.ios.ios
    aruba_network_os_values:
      - aruba
      - community.network.aruba
      - arubaoss
      - aruba-os-switch
      - arubaos-switch
      - aos-switch
    aruba_aoscx_network_os_values:
      - aoscx
      - aruba-cx
      - arubacx
      - arubaos-cx
      - arubanetworks.aoscx.aoscx

  tasks:
    - name: 0.1. Xác định network OS của thiết bị
      set_fact:
        nb_sync_network_os: "{{ ansible_network_os | default('cisco.ios.ios') | string | lower }}"

    - name: 0.2. Chọn template và platform deploy phù hợp
      set_fact:
        nb_sync_platform: "{{ 'cisco' if nb_sync_network_os in cisco_network_os_values else 'aruba_aoscx' if nb_sync_network_os in aruba_aoscx_network_os_values else 'aruba_os_switch' if nb_sync_network_os in aruba_network_os_values else 'unsupported' }}"
        nb_sync_template: "{{ 'ios/netbox_sync.j2' if nb_sync_network_os in cisco_network_os_values else 'aoscx/netbox_sync.j2' if nb_sync_network_os in aruba_aoscx_network_os_values else 'aruba/netbox_sync.j2' if nb_sync_network_os in aruba_network_os_values else '' }}"
        nb_sync_config_path: "{{ playbook_dir }}/../generated_configs/{{ inventory_hostname }}_netbox_sync.cfg"

    - name: 0.3. Dừng nếu thiết bị chưa được hỗ trợ
      assert:
        that:
          - nb_sync_platform != 'unsupported'
        fail_msg: >-
          Chưa hỗ trợ ansible_network_os={{ ansible_network_os | default('undefined') }}
          cho {{ inventory_hostname }}. Hiện hỗ trợ Cisco IOS, ArubaOS-Switch và Aruba AOS-CX.

    - name: 0.4. Dừng nếu inventory đang trỏ nhầm về chính máy automation
      assert:
        that:
          - ansible_host | default('') not in lookup('pipe', 'hostname -I').split()
        fail_msg: >-
          {{ inventory_hostname }} đang có ansible_host={{ ansible_host | default('undefined') }},
          trùng IP của máy automation. Hãy sửa primary management IP trong NetBox trước khi deploy.
      delegate_to: localhost

    # ---------------------------------------------------------
    # BƯỚC 1: KÉO DỮ LIỆU TỪ NETBOX QUA API (Sử dụng nb_lookup)
    # ---------------------------------------------------------
    - name: 1.1. Lấy toàn bộ danh sách VLAN (từ nhóm SCH_VLAN_Group)
      set_fact:
        nb_vlans: "{{ query('netbox.netbox.nb_lookup', 'vlans', api_endpoint=netbox_url, token=netbox_token) }}"
      delegate_to: localhost

    - name: 1.2. Lấy chi tiết toàn bộ Interface của chính Switch này
      set_fact:
        nb_interfaces: "{{ query('netbox.netbox.nb_lookup', 'interfaces', api_endpoint=netbox_url, token=netbox_token, api_filter='device=' ~ inventory_hostname) }}"
      delegate_to: localhost

    # ---------------------------------------------------------
    # BƯỚC 2: KẾT HỢP DỮ LIỆU VỚI TEMPLATE ĐỂ RENDER CẤU HÌNH
    # ---------------------------------------------------------
    - name: 2. Render cấu hình CLI hoàn chỉnh lưu vào thư mục generated_configs
      template:
        src: "{{ playbook_dir }}/../templates/{{ nb_sync_template }}"
        dest: "{{ nb_sync_config_path }}"
      delegate_to: localhost

    # ---------------------------------------------------------
    # BƯỚC 3: ĐẨY CẤU HÌNH XUỐNG SWITCH THẬT
    # ---------------------------------------------------------
    - name: 3. Đẩy cấu hình từ file đã render xuống Switch Cisco
      cisco.ios.ios_config:
        src: "{{ nb_sync_config_path }}"
        backup: yes
        backup_options:
          filename: "{{ inventory_hostname }}_pre_netbox.bak"
          dir_path: "{{ playbook_dir }}/../backups"
      when: nb_sync_platform == 'cisco'

    - name: 3. Đẩy cấu hình từ file đã render xuống Switch ArubaOS-Switch
      community.network.aruba_config:
        src: "{{ nb_sync_config_path }}"
        backup: yes
        backup_options:
          filename: "{{ inventory_hostname }}_pre_netbox.bak"
          dir_path: "{{ playbook_dir }}/../backups"
      when: nb_sync_platform == 'aruba_os_switch'

    - name: 3. Đẩy cấu hình từ file đã render xuống Switch Aruba AOS-CX
      arubanetworks.aoscx.aoscx_config:
        src: "{{ nb_sync_config_path }}"
        backup: yes
        backup_options:
          filename: "{{ inventory_hostname }}_pre_netbox.bak"
          dir_path: "{{ playbook_dir }}/../backups"
      vars:
        ansible_become: no
      when: nb_sync_platform == 'aruba_aoscx'
